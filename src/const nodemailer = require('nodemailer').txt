const nodemailer = require('nodemailer');

async function sendMail({ to, subject, text }) {
  let transporter = nodemailer.createTransport({
    service: 'gmail',
    auth: {
      user: 'bhagvaty83@gmail.com', // replace with your Gmail
      pass: '123456',   // use an App Password, not your Gmail password
    },
  });

  let info = await transporter.sendMail({
    from: '"Your Name" <bhagvaty83@gmail.com>',
    to,
    subject,
    text,
  });

  return info;
}

module.exports = sendMail;
